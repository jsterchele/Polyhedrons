  3     /**
  4      * Create scene, camera, renderer.
  5      */
  6     var setUp = function() {
  7         var result = {};
  8 
  9         // create a scene, that will hold all our elements such
 10         // as objects, cameras and lights.
 11         var scene = new THREE.Scene();
 12 
 13         // create a camera, which defines where we're looking at.
 14         // a frustrum is a pyramid with a square base the apex cut off
 15         // 45 is vertical field of view of the frustrum
 16         // window.innerWidth/window.innerHeight is aspect ratio of the frustrum
 17         // 0.1 is the distance to the near plane of the frustrum
 18         // 1000 is the distance to the far plane of the frustrum
 19         var camera = new THREE.PerspectiveCamera(45,
 20             window.innerWidth / window.innerHeight, 0.1, 1000);
 21 
 22         var orbitControls = new THREE.OrbitControls( camera );
 23         orbitControls.autoRotate = false;
 24         var clock = new THREE.Clock();
 25 
 26         // create a render and set the size
 27         var renderer = new THREE.WebGLRenderer();
 28         // specify the color of the background for this image
 29         renderer.setClearColor(new THREE.Color(0xDDFFCC));
 30         renderer.setSize(window.innerWidth, window.innerHeight);
 31         renderer.shadowMapEnabled = true;
 32 
 33         // specify the lengths of the 3 axes
 34         var axes = new THREE.AxisHelper(24);
 35         scene.add(axes);
 36 
 37         // position and point the camera to the center of the scene
 38         camera.position.x = 0;
 39         camera.position.y = 0;
 40         camera.position.z = 300;
 41         camera.lookAt(scene.position);
 42 
 43         var origin = new THREE.Object3D();
 44         origin.position = new THREE.Vector3( 0, 0, 0 );
 45 
 46         result.camera = camera;
 47         result.clock = clock;
 48         result.orbitControls = orbitControls;
 49         result.origin = origin;
 50         result.renderer = renderer;
 51         result.scene = scene;
 52 
 53         return result;
 54     }; // setUp()

 57     /**
 58      * Give the program the means to resize the image
 59      * when a user resizes the window that contains
 60      * the image.
 61      *
 62      * @param {THREE.PerspectiveCamera} camera (could also be a different kind of camera)
 63      * @param {THREE.WebGLRenderer} renderer (could also be a different kind of renderer)
 64      */
 65     var makeResizable = function( camera, renderer ) {
 66         /**
 67          * Resize image when user resizes the window.
 68          */
 69         var onResize = function() {
 70             camera.aspect = window.innerWidth / window.innerHeight;
 71             camera.updateProjectionMatrix();
 72             renderer.setSize( window.innerWidth, window.innerHeight );
 73         }; // onResize()
 74 
 75         window.addEventListener( 'resize', onResize, false );
 76 
 77     }; // makeResizable()

286     /**
287      * Adds spheres to the vertexs, and adds them to the scene
288      * Does NOT print any vertices that have a "mid" attribute that is true
289      * @param geometry { THREE.Geometry } a THREE.Geometry object.
290      * @param scene { THREE.Scene } a THREE.Scene object
291      * @param scale a scale factor for the shape
292      **/
293     function vertexSpheres( geometry, scene, scale )
294     {
295         var vertSpheres = [];
296         var sGeometry = new THREE.SphereGeometry( 1, 32, 32 );
297         var sMaterial = new THREE.MeshBasicMaterial( {color: 0x695959} );
298         for( var i = 0; i < geometry.vertices.length; i++)
299         {
300           if( geometry.vertices[i].mid != true)
301           {
302           var sphere = new THREE.Mesh( sGeometry, sMaterial );
303           sphere.position.x = geometry.vertices[i].x * scale;
304           sphere.position.y = geometry.vertices[i].y * scale;
305           sphere.position.z = geometry.vertices[i].z * scale;
306           vertSpheres.push( sphere );
307           scene.add( sphere );
308           }
309         }
310     }
